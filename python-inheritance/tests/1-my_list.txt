======================
How to Use 1-mylist.py
======================

    This module defines a list of integers using the `MyList` class.

===========================
# main file: 
===========================
    #!/usr/bin/python3
    >>> MyList = __import__('1-my_list').MyList

    >>> my_list = MyList()
    >>> my_list.append(1)
    >>> my_list.append(4)
    >>> my_list.append(2)
    >>> my_list.append(3)
    >>> my_list.append(5)
    >>> print(my_list)
    [1, 4, 2, 3, 5]
    >>> my_list.print_sorted()
    [1, 2, 3, 4, 5]
    >>> print(my_list)
    [1, 4, 2, 3, 5]

===========================
# negative integer:
===========================
    >>> my_list = MyList()
    >>> print(my_list)
    []
    >>> my_list.append(-1)
    >>> print(my_list)
    [-1]
    >>> my_list.append(-4)
    >>> print(my_list)
    [-1, -4]
    >>> my_list.append(-2)
    >>> print(my_list)
    [-1, -4, -2]
    >>> my_list.append(-3)
    >>> print(my_list)
    [-1, -4, -2, -3]
    >>> my_list.append(-5)
    >>> print(my_list)
    [-1, -4, -2, -3, -5]
    >>> print(my_list)
    [-1, -4, -2, -3, -5]
    >>> my_list.print_sorted()
    [-5, -4, -3, -2, -1]
    >>> print(my_list)
    [-1, -4, -2, -3, -5]


=========================
# positive integer:
=========================
    >>> my_list = MyList()
    >>> print(my_list)
    []
    >>> my_list.append(1)
    >>> print(my_list)
    [1]
    >>> my_list.append(4)
    >>> print(my_list)
    [1, 4]
    >>> my_list.append(2)
    >>> print(my_list)
    [1, 4, 2]
    >>> my_list.append(3)
    >>> print(my_list)
    [1, 4, 2, 3]
    >>> my_list.append(5)
    >>> print(my_list)
    [1, 4, 2, 3, 5]
    >>> print(my_list)
    [1, 4, 2, 3, 5]
    >>> my_list.print_sorted()
    [1, 2, 3, 4, 5]
    >>> print(my_list)
    [1, 4, 2, 3, 5]
